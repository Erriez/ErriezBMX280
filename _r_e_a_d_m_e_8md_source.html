10 ## Arduino library features
11 
12 - Measurements:
13  - BMP280: Temperature / pressure / approximate altitude
14  - BME280: Temperature / pressure / approximate altitude / humidity
15 - Three operation modes:
16  - Normal mode: Continues conversion
17  - Forced mode: One-time conversion
18  - Standby mode: Low-power, no conversion
19 - Sampling configuration
20 - Chip detect / read chip ID
21 - I2C interface only
22 - Small flash/RAM footprint
23 
24 
25 ## BMP280/BME280 sensor specifications
26 
27 - Operating voltage: 1.71V .. 3.6V max
28 - Low current:
29  - 1.8 uA @ 1 Hz humidity and temperature
30  - 2.8 uA @ 1 Hz pressure and temperature
31  - 3.6 uA @ 1 Hz humidity, pressure and temperature
32  - 0.1 uA in sleep mode
33 - Operating range: -40...+85 °C, 0...100 % rel. humidity, 300...1100 hPa
34 - I2C bus interface: max 3.4 MHz
35 - No additional electronic components needed

69 ## Example
70 
71 ```c++
72 #include <Wire.h>
73 #include <ErriezBMX280.h>
74 
75 // Adjust sea level for altitude calculation
76 #define SEA_LEVEL_PRESSURE_HPA 1026.25
77 
78 // Create BMX280 object I2C address 0x76 or 0x77
79 ErriezBMX280 bmx280 = ErriezBMX280(0x76);
80 
81 
82 void setup()
83 {
84  // Initialize serial
85  delay(500);
86  Serial.begin(115200);
87  while (!Serial) {
88  ;
89  }
90  Serial.println(F("\nErriez BMP280/BMX280 example"));
91 
92  // Initialize I2C bus
93  Wire.begin();
94  Wire.setClock(400000);
95 
96  // Initialize sensor
97  while (!bmx280.begin()) {
98  Serial.println(F("Error: Could not detect sensor"));
99  delay(3000);
100  }
101 
102  // Print sensor type
103  Serial.print(F("\nSensor type: "));
104  switch (bmx280.getChipID()) {
105  case CHIP_ID_BMP280:
106  Serial.println(F("BMP280\n"));
107  break;
108  case CHIP_ID_BME280:
109  Serial.println(F("BME280\n"));
110  break;
111  default:
112  Serial.println(F("Unknown\n"));
113  break;
114  }
115 }
116 
117 void loop()
118 {
119  Serial.print(F("Temperature: "));
120  Serial.print(bmx280.readTemperature());
121  Serial.println(" C");
122 
123  if (bmx280.getChipID() == CHIP_ID_BME280) {
124  Serial.print(F("Humidity: "));
125  Serial.print(bmx280.readHumidity());
126  Serial.println(" %");
127  }
128 
129  Serial.print(F("Pressure: "));
130  Serial.print(bmx280.readPressure() / 100.0F);
131  Serial.println(" hPa");
132 
133  Serial.print(F("Altitude: "));
134  Serial.print(bmx280.readAltitude(SEA_LEVEL_PRESSURE_HPA));
135  Serial.println(" m");
136 
137  Serial.println();
138 
139  delay(1000);
140 }
141 ```

161 ### Set sampling
162 
163 The sensor sampling and mode can be configured with function `setSampling()`. Recommended modes of
164 operation according to the datasheet chapter "Recommended modes of operation":
165 
166 ```c++
167 // Set sampling
168 //
169 // Weather
170 // - forced mode, 1 sample / minute
171 // - pressure ×1, temperature ×1, humidity ×1
172 // - filter off
173 //
174 // Humidity sensing
175 // - forced mode, 1 sample / second
176 // - pressure ×0, temperature ×1, humidity ×1
177 // - filter off
178 //
179 // Indoor navigation
180 // - normal mode, t standby = 0.5 ms
181 // - pressure ×16, temperature ×2, humidity ×1
182 // - filter coefficient 16
183 //
184 // Gaming
185 // - forced mode, t standby = 0.5 ms
186 // - pressure ×1, temperature ×1, humidity ×1
187 // - filter off
188 bmx280.setSampling(BMX280_MODE_NORMAL, // SLEEP, FORCED, NORMAL
189  BMX280_SAMPLING_X16, // Temp: NONE, X1, X2, X4, X8, X16
190  BMX280_SAMPLING_X16, // Press: NONE, X1, X2, X4, X8, X16
191  BMX280_SAMPLING_X16, // Hum: NONE, X1, X2, X4, X8, X16 (BME280)
192  BMX280_FILTER_X16, // OFF, X2, X4, X8, X16
193  BMX280_STANDBY_MS_500);// 0_5, 10, 20, 62_5, 125, 250, 500, 1000
194  ```
